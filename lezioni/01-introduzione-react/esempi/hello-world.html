3. Componente con Stato (useState)

Un componente che gestisce lo stato interno:

function Counter() {
  const [count, setCount] = React.useState(0);
  return (
    <div>
      <p>Contatore: <span class="highlight">{count}</span></p>
      <button onClick={() => setCount(count + 1)}>+1</button>
      <button onClick={() => setCount(count - 1)}>-1</button>
      <button onClick={() => setCount(0)}>Reset</button>
    </div>
  );
}

4. Componente Interattivo

Un componente che risponde alle interazioni dell'utente:

function InteractiveDemo() {
  const [message, setMessage] = React.useState('Clicca i pulsanti!');
  const [color, setColor] = React.useState('#61dafb');
  
  return (
    <div>
      <h3 style={{color: color}}>{message}</h3>
      <button onClick={() => setMessage('Hai cliccato il pulsante blu!')}>
        Messaggio Blu
      </button>
      <button onClick={() => setColor('#ff6b6b')}>
        Cambia Colore
      </button>
    </div>
  );
}
